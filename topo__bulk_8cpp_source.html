154  std::vector<int> peripheralRings; // Indices which may be peripheral rings
155  bool cond1, cond2, cond3; // Conditions for DDC

157  std::vector<int> DDCRings; // Indices of rings which constitute a single DDC,
158  // with the equatorial ring first
159

161  // the hexagonal rings
162  for (int iring = 0; iring < totalRingNum; iring++) {
163  peripheralRings.clear();
164  // ------------
165  // Step one: Find all rings which contain each index (m_k) of the equatorial

167  cond1 = ring::conditionOneDDC(rings, &peripheralRings, iring);
168  if (cond1 == false) {

170  }

595 bool ring::basalConditions(std::vector<std::vector<int>> nList,
596  std::vector<int> *basal1, std::vector<int> *basal2) {
597  int l1 = (*basal1)[0]; // first element of basal1 ring
598  int l2 = (*basal1)[1]; // second element of basal1 ring
599  int ringSize = 6; // Size of the ring; each ring contains 6 elements
600  int m_k; // Atom Index (in pointCloud) of element in basal2
601  int kIndex; // Index of m_k in basal2, corresponding to m_k
602  int currentKindex; // Current k index when finding alternating elements of
603  // basal2
604  std::vector<int> evenTriplet; // contains m_k, m_{k+2}, m_{k+4}
605  std::vector<int> oddTriplet; // contains m_{k+1}, m_{k+3}, m_{k+5}
606  int compare1, compare2; // l3 and l5 OR l4 and l6
607  int index;
608  bool l1_neighbour, l2_neighbour; // m_k is a neighbour of l1(true) or not
609  // (false); m_k is a neighbour of l2(true)
610  bool isNeigh, notNeigh; // Used to check if the rings are basal or not
611 
612  // ---------------------------------------------
613  // SEARCH FOR L1_NEIGHBOUR OR L2_NEIGHBOUR
614  // Search for whether an element of basal2 is a neighbour of l1 or l2
615  for (int k = 0; k < ringSize; k++) {
616  // init
617  l1_neighbour = false;
618  l2_neighbour = false;
619  m_k = (*basal2)[k];
620 
621  // ---------------
622  // CHECK IF M_K MATCHES L1 NEIGHBOURS
623  auto it1 = std::find(nList[l1].begin() + 1, nList[l1].end(), m_k);
624  // If m_k was found in l1's nList
625  if (it1 != nList[l1].end()) {
626  compare1 = (*basal1)[2]; // l3
627  compare2 = (*basal1)[4]; // l5
628  kIndex = k; // Saving the array index of m_k
629  l1_neighbour = true;
630  break;
631  } // m_k found in l1's nList
632  // ---------------
633  // CHECK IF M_K MATCHES L2 NEIGHBOURS
634  auto it2 = std::find(nList[l2].begin() + 1, nList[l2].end(), m_k);
635  // If m_k was found in l1's nList
636  if (it2 != nList[l2].end()) {
637  compare1 = (*basal1)[3]; // l4
638  compare2 = (*basal1)[5]; // l6
639  kIndex = k; // Saving the array index of m_k
640  l2_neighbour = true;
641  break;
642  } // m_k found in l1's nList
643  // ---------------
644  } // End of search for basal2 elements in l1 or l2's nList
645  // ---------------------------------------------
646 
647  // Return false if neither l1 nor l2 have any neighbours
648  // in basal2
649 
650  if (l1_neighbour == false && l2_neighbour == false) {
651  return false;
652  } // basal conditions not fulfilled
653 
654  // Get the alternating elements starting with kIndex.
655  // 'evenTriplet': m_k, m_{k+2}, m_{k+4} - neighbours of compare1 and compare2.
656  // 'oddTriplet': m_{k+1}, m_{k+3}, m_{k+5}- cannot be neighbours of basal1
657  for (int k = 0; k <= 5; k++) {
658  currentKindex = kIndex + k; // k
659  // Wrap-around
660  if (currentKindex >= ringSize) {
661  currentKindex -= ringSize;
662  } // end of wrap-around of k
663  //
664  // Update 'evenTriplet'
665  if (k % 2 == 0) {
666  evenTriplet.push_back((*basal2)[currentKindex]);
667  } // end of update of evenTriplet
668  // Update 'oddTriplet'
669  else {
670  oddTriplet.push_back((*basal2)[currentKindex]);
671  } // end of update of oddTriplet
672  } // End of getting alternating triplets
673 
674  // ---------------------------------------------
675  // CONDITION1: m_{k+2} and m_{k+4} must be bonded to l3 and l5 (if l1 is a
676  // neighbour) or m_{k+2} and m_{k+4} must be bonded to l4 and l6 (if l2 is a
677  // neighbour) Basically, this boils down to checking whether compare1 and
678  // compare2 are in the neighbour lists of the last two elements of evenTriplet
679 
680  isNeigh = ring::basalNeighbours(nList, &evenTriplet, compare1, compare2);
681 
682  // If condition1 is not true, then the candidate
683  // rings are not part of an HC
684  if (!isNeigh) {
685  return false;
686  } // Not an HC
687 
688  // ---------------------------------------------
689  // CONDITION2: m_{k+1}, m_{k+3} and m_{k+5} must NOT be bonded to any element
690  // in basal1.
691  // Basically, this boils down to checking whether the elements of oddTriplet
692  // are in the neighbour lists of all the elements of basal1.
693 
694  // condition 2. This must be true for an HC
695  notNeigh = ring::notNeighboursOfRing(nList, &oddTriplet, basal1);
696 
697  // If condition2 is not true, the the candidate rings
698  // are not part of an HC
699  if (!notNeigh) {
700  return false;
701  } // Not an HC
702 
703  // Otherwise, all the conditions are true and this is an HC
704  return true;
705
